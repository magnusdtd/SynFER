\section{Phân tích và Đánh giá}
\begin{frame}{Đối chiếu phương pháp truyền thống và hiện đại}
    \textbf{Điểm tương đồng:} Đều xoay quanh hệ tư tưởng rằng biểu cảm là kết quả trực tiếp của chuyển động cơ mặt cục bộ.
    \vspace{0.3cm}

    \textbf{Sự tiến hóa:}
    \begin{itemize}
        \item \textbf{Truyền thống:} Sử dụng thuật toán lưới điểm học đơn giản. Dễ gây rách lưới, biến dạng ở các biên nếu khuôn mặt mục tiêu khác biệt với khuôn mặt nguồn.
        \item \textbf{Hiện đại (Mô hình Diffusion - SynFER):} Không trực tiếp biến đổi trên điểm ảnh mà sinh ảnh hoàn toàn mới từ nhiễu. Tái hiện được thay đổi quang học phức tạp.
    \end{itemize}
\end{frame}
